local aimLockEnabled = false
local aimRadius = 1000  -- Bán kính quét mục tiêu
local aimSpeed = 0.15  -- Tốc độ điều chỉnh góc nhìn
local selectedTarget = nil
local crosshairGui = nil  -- GUI tâm ngắm

-- **🔹 Tạo GUI tâm ngắm**
local function createCrosshair()
    if crosshairGui then return end  -- Nếu đã tạo, không tạo lại

    local gui = Instance.new("ScreenGui")
    gui.Parent = game.Players.LocalPlayer:FindFirstChildOfClass("PlayerGui") or game.CoreGui
    gui.Name = "Crosshair"

    local crosshair = Instance.new("Frame")
    crosshair.Size = UDim2.new(0, 20, 0, 20)  -- Kích thước tâm
    crosshair.Position = UDim2.new(0.5, -10, 0.5, -10)  -- Giữ giữa màn hình
    crosshair.BackgroundColor3 = Color3.fromRGB(0, 255, 0)  -- Màu xanh lá
    crosshair.BackgroundTransparency = 0.2
    crosshair.Parent = gui

    crosshairGui = gui
end

-- **🔹 Xóa GUI tâm ngắm**
local function removeCrosshair()
    if crosshairGui then
        crosshairGui:Destroy()
        crosshairGui = nil
    end
end

-- **🔹 Xác định quái/NPC hợp lệ (Không nhắm vào player)**
local function isValidTarget(entity)
    if entity:IsA("Model") and entity:FindFirstChild("Humanoid") and entity:FindFirstChild("HumanoidRootPart") then
        if game.Players:GetPlayerFromCharacter(entity) then  -- Bỏ qua Player
            return false
        end
        return true
    end
    return false
end

-- **🔹 Tìm quái gần nhất**
local function findClosestTarget()
    local player = game.Players.LocalPlayer
    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return nil end

    local closestTarget = nil
    local shortestDistance = aimRadius

    for _, entity in pairs(game.Workspace:GetChildren()) do
        if isValidTarget(entity) then
            local distance = (character.HumanoidRootPart.Position - entity.HumanoidRootPart.Position).Magnitude
            if distance < shortestDistance then
                closestTarget = entity
                shortestDistance = distance
            end
        end
    end

    return closestTarget
end

-- **🔹 Camlock tự động nhắm vào đầu mục tiêu**
local function aimAtTarget(target)
    if target and target:FindFirstChild("HumanoidRootPart") then
        local player = game.Players.LocalPlayer
        local camera = game.Workspace.CurrentCamera
        local headPosition = target.HumanoidRootPart.Position + Vector3.new(0, 2.5, 0)  -- Vị trí đầu mục tiêu
        local newCFrame = CFrame.new(camera.CFrame.Position, headPosition)
        camera.CFrame = camera.CFrame:Lerp(newCFrame, aimSpeed)
    end
end

-- **🔹 Kích hoạt Camlock + Tâm Ngắm**
local function startAimLock()
    aimLockEnabled = not aimLockEnabled

    if aimLockEnabled then
        print("🎯 Camlock BẬT (Ghim Đầu + Hiện Tâm)")
        createCrosshair()  -- Hiện tâm ngắm

        spawn(function()
            while aimLockEnabled do
                selectedTarget = findClosestTarget()
                if selectedTarget then
                    aimAtTarget(selectedTarget)
                end
                wait(0.05)  -- Kiểm soát tần suất quét mục tiêu
            end
        end)
    else
        print("❌ Camlock TẮT")
        removeCrosshair()  -- Ẩn tâm ngắm
        selectedTarget = nil
    end
end

-- **🔹 Thêm nút vào menu**
AddButton(Tab1o, {
    Name = "🎯AIMBOT ENTITY",
    Callback = function()
        startAimLock()
    end
})
