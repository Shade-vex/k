local espEnabled = false
local espColor = Color3.fromRGB(0, 255, 0) -- MÃ u xanh lÃ¡
local textSize = 15 -- KÃ­ch thÆ°á»›c chá»¯ cá»‘ Ä‘á»‹nh
local updateRate = 1 -- Cáº­p nháº­t má»—i 1 giÃ¢y
local maxDistance = 1500 -- BÃ¡n kÃ­nh 1500m

-- **ğŸ”¹ XÃ¡c Ä‘á»‹nh item cÃ³ thá»ƒ nháº·t**
local function isValidItem(item)
    if item:IsA("Tool") or item:IsA("Accessory") or item:IsA("BackpackItem") then
        return true
    end
    -- **Cháº·n cÃ¡c váº­t thá»ƒ lá»›n khÃ´ng pháº£i item**
    local blacklist = {"Tree", "Building", "Wall", "Track", "Bridge", "Platform"}
    for _, name in pairs(blacklist) do
        if item.Name:lower():find(name:lower()) then
            return false
        end
    end
    return false
end

-- **ğŸ”¹ HÃ m táº¡o ESP cho item**
local function createESP(item)
    local basePart = item:FindFirstChildWhichIsA("BasePart") or item
    if basePart and not basePart:FindFirstChild("EspLabel") then
        local billboard = Instance.new("BillboardGui")
        billboard.Name = "EspLabel"
        billboard.Adornee = basePart
        billboard.Size = UDim2.new(0, 200, 0, 50)
        billboard.StudsOffset = Vector3.new(0, 2, 0)
        billboard.AlwaysOnTop = true

        local textLabel = Instance.new("TextLabel", billboard)
        textLabel.Size = UDim2.new(1, 0, 1, 0)
        textLabel.BackgroundTransparency = 1
        textLabel.TextColor3 = espColor
        textLabel.TextSize = textSize
        textLabel.Font = Enum.Font.SourceSansBold
        textLabel.TextStrokeTransparency = 0.5
        textLabel.TextScaled = false

        billboard.Parent = basePart
    end
end

-- **ğŸ”¹ Cáº­p nháº­t ESP cho toÃ n bá»™ item má»—i 5 giÃ¢y**
local function updateESP()
    while espEnabled do
        pcall(function()
            local player = game.Players.LocalPlayer
            if player and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                local playerPos = player.Character.HumanoidRootPart.Position

                for _, item in pairs(game.Workspace:GetChildren()) do
                    if isValidItem(item) then
                        local basePart = item:FindFirstChildWhichIsA("BasePart") or item
                        if basePart then
                            local distance = (playerPos - basePart.Position).Magnitude
                            if distance <= maxDistance then
                                createESP(basePart)
                                local label = basePart:FindFirstChild("EspLabel")
                                if label then
                                    local textLabel = label:FindFirstChildOfClass("TextLabel")
                                    if textLabel then
                                        textLabel.Text = string.format("%s\n[%dm]", item.Name, distance)
                                    end
                                end
                            else
                                if basePart:FindFirstChild("EspLabel") then
                                    basePart.EspLabel:Destroy()
                                end
                            end
                        end
                    end
                end
            end
        end)
        wait(updateRate)
    end
end

-- **ğŸ”¹ Báº­t/Táº¯t ESP**
local function toggleESP()
    espEnabled = not espEnabled
    if espEnabled then
        print("ğŸŸ¢ ESP Item: Báº¬T (Cáº­p nháº­t má»—i 5 giÃ¢y)")
        updateESP()
    else
        print("ğŸ”´ ESP Item: Táº®T")
        for _, item in pairs(game.Workspace:GetChildren()) do
            if isValidItem(item) then
                local basePart = item:FindFirstChildWhichIsA("BasePart") or item
                if basePart and basePart:FindFirstChild("EspLabel") then
                    basePart.EspLabel:Destroy()
                end
            end
        end
    end
end
