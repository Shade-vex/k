local attackEnabled = false
local attackRate = 1  -- Mỗi giây tấn công 1 lần
local damageAmount = 50 -- Sát thương mỗi đòn đánh
local attackRange = 1000 -- Bán kính tấn công

-- **🔹 Tìm quái hợp lệ**
local function isValidEnemy(entity)
    if entity:IsA("Model") and entity:FindFirstChild("Humanoid") and entity:FindFirstChild("HumanoidRootPart") then
        -- Không tấn công người chơi
        if game.Players:GetPlayerFromCharacter(entity) then
            return false
        end
        return true
    end
    return false
end

-- **🔹 Kiểm tra khoảng cách**
local function isInRange(entity)
    local player = game.Players.LocalPlayer
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        local distance = (player.Character.HumanoidRootPart.Position - entity.HumanoidRootPart.Position).Magnitude
        return distance <= attackRange
    end
    return false
end

-- **🔹 Gửi lệnh đánh thật đến server**
local function attackEnemies()
    while attackEnabled do
        pcall(function()
            for _, entity in pairs(game.Workspace:GetChildren()) do
                if isValidEnemy(entity) and isInRange(entity) then
                    local humanoid = entity:FindFirstChild("Humanoid")
                    if humanoid and humanoid.Health > 0 then
                        -- **Gửi lệnh tấn công đến server (chính xác)**
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("DamageEnemy", entity, damageAmount)

                        -- **Làm quái mất HP thật**
                        humanoid.Health = humanoid.Health - damageAmount
                    end
                end
            end
        end)
        wait(attackRate)
    end
end

-- **🔹 Bật/Tắt Kill Aura**
local function toggleKillAura()
    attackEnabled = not attackEnabled
    if attackEnabled then
        print("🔥 Kill Aura: BẬT (50 HP/s, 1000m)")
        attackEnemies()
    else
        print("❌ Kill Aura: TẮT")
    end
end

-- **🔹 Thêm nút vào menu**
AddButton(Tab1o, {
    Name = "⚔️KILL AURA",
    Callback = function()
        toggleKillAura()
    end
})

local speedEnabled = false
local speedValue = 75
local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:FindFirstChildOfClass("Humanoid")

-- 🛡 Chặn Anti-Cheat hoàn toàn (Bypass mạnh hơn)
local function bypassAntiCheat()
    for _, obj in pairs(getgc(true)) do
        if type(obj) == "function" and islclosure(obj) then
            local info = debug.getinfo(obj)
            if info.source and string.find(info.source, "AntiCheat") then
                setreadonly(obj, false)
                disablefunction(obj)
                print("🚫 Đã vô hiệu hóa Anti-Cheat!")
            end
        end
    end
end

-- 🎭 Fake Admin quyền đặc biệt (Để server không kiểm tra tốc độ)
local function fakeAdmin()
    if sethiddenproperty then
        sethiddenproperty(player, "SimulationRadius", math.huge)
    end
    if setfflag then
        setfflag("HumanoidParallelRemoveNoPhysics", "False")
        setfflag("HumanoidParallelRemoveNoPhysicsNoSimulate2", "False")
    end
    print("✅ Đã Fake Admin thành công!")
end

-- 🚀 Hàm thiết lập tốc độ (Bypass mạnh hơn)
local function setSpeed()
    while speedEnabled do
        if humanoid then
            humanoid.WalkSpeed = speedValue
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Climbing, false) -- Ngăn check địa hình
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false) -- Ngăn bug tốc độ
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Physics, false) -- Ngăn vật lý gây gián đoạn
        end
        wait(0.1)
    end
    humanoid.WalkSpeed = 16 -- Trả về tốc độ mặc định khi tắt
end

-- 🟢 Nút bật/tắt Fake Admin + Tăng tốc hoàn toàn
AddButton(Tab1o, {
    Name = "🚀 SPEED OP",
    Callback = function()
        speedEnabled = not speedEnabled
        if speedEnabled then
            fakeAdmin()
            bypassAntiCheat()
            setSpeed()
        else
            humanoid.WalkSpeed = 16 -- Trả lại tốc độ mặc định khi tắt
        end
    end
})
