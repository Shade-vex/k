function updateCircle(Fov, colorCircle)
    if circle.__OBJECT_EXISTS then
        circle.Transparency = 1
        circle.Visible = true
        circle.Thickness = 2
        circle.Color = colorCircle
        circle.NumSides = 100
        circle.Radius = (Fov * 6) / 2
        circle.Filled = false
        circle.Position = Vector2.new(mouse.X, mouse.Y + (guiservice:GetGuiInset().Y))
    end
end

-- Hook v√†o RemoteEvent ƒë·ªÉ t·ª± ƒë·ªông nh·∫Øm
spawn(function()
    local gg = getrawmetatable(game)
    local old = gg.__namecall
    setreadonly(gg, false)
    gg.__namecall = newcclosure(function(...)
        local method = getnamecallmethod()
        local args = { ... }
        if tostring(method) == "FireServer" and tostring(args[1]) == "RemoteEvent" then
            if AimBotEnabled and Playersaimbot then
                args[2] = PlayersPosition
                return old(unpack(args))
            end
        end
        return old(...)
    end)
end)

local hitboxEnabled = false  -- Bi·∫øn ƒë·ªÉ b·∫≠t/t·∫Øt hitbox
local hitboxSize = Vector3.new(70, 70, 70)  -- K√≠ch th∆∞·ªõc hitbox x70
local transparency = 0.5  -- ƒê·ªô trong su·ªët c·ªßa hitbox (0 l√† kh√¥ng trong su·ªët, 1 l√† v√¥ h√¨nh)
local hitboxLoopRunning = false  -- Bi·∫øn ki·ªÉm so√°t v√≤ng l·∫∑p ki·ªÉm tra hitbox

-- H√†m ch·ªânh hitbox c·ªßa ng∆∞·ªùi ch∆°i kh√°c
local function setHitbox(player)
    -- Ki·ªÉm tra kh√¥ng √°p d·ª•ng hitbox cho ch√≠nh m√¨nh
    if player == game.Players.LocalPlayer then return end  

    local character = player.Character
    if character and character:FindFirstChild("HumanoidRootPart") then
        local rootPart = character.HumanoidRootPart
        rootPart.Size = hitboxSize
        rootPart.Transparency = transparency
        rootPart.CanCollide = false
    end
end

-- H√†m ki·ªÉm tra v√† c·∫≠p nh·∫≠t hitbox cho t·∫•t c·∫£ ng∆∞·ªùi ch∆°i trong game
local function checkAndSetHitboxes()
    if hitboxEnabled then
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character then
                -- Ki·ªÉm tra v√† set hitbox cho player
                setHitbox(player)
            end
        end
    end
end

-- H√†m b·∫≠t/t·∫Øt hitbox
local function toggleHitbox()
    hitboxEnabled = not hitboxEnabled
    if hitboxEnabled then
        -- √Åp d·ª•ng hitbox cho t·∫•t c·∫£ player hi·ªán t·∫°i (tr·ª´ LocalPlayer)
        checkAndSetHitboxes()

        -- K·∫øt n·ªëi s·ª± ki·ªán khi c√≥ player m·ªõi v√†o game v√† sau 3s s·∫Ω thay ƒë·ªïi hitbox
        game.Players.PlayerAdded:Connect(function(player)
            player.CharacterAdded:Connect(function()
                wait(3)  -- Ch·ªù 3 gi√¢y ƒë·ªÉ nh√¢n v·∫≠t load xong
                setHitbox(player)
            end)
        end)

        -- K·∫øt n·ªëi s·ª± ki·ªán khi player h·ªìi sinh
        game.Players.PlayerAdded:Connect(function(player)
            player.CharacterAdded:Connect(function()
                wait(3)  -- Ch·ªù 3 gi√¢y ƒë·ªÉ nh√¢n v·∫≠t load xong
                setHitbox(player)
            end)
        end)

        showRainbowText()  -- Hi·ªÉn th·ªã ch·ªØ khi b·∫≠t hitbox

        -- B·∫Øt ƒë·∫ßu v√≤ng l·∫∑p ki·ªÉm tra hitbox m·ªói gi√¢y
        hitboxLoopRunning = true
        while hitboxLoopRunning do
            wait(1)  -- Ki·ªÉm tra m·ªói gi√¢y
            checkAndSetHitboxes()  -- Ki·ªÉm tra v√† set hitbox cho t·∫•t c·∫£ player
        end
    else
        -- D·ª´ng v√≤ng l·∫∑p v√† reset hitbox cho t·∫•t c·∫£ player
        hitboxLoopRunning = false
        for _, player in pairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer then
                local character = player.Character
                if character and character:FindFirstChild("HumanoidRootPart") then
                    local rootPart = character.HumanoidRootPart
                    rootPart.Size = Vector3.new(2, 2, 1)  -- K√≠ch th∆∞·ªõc m·∫∑c ƒë·ªãnh
                    rootPart.Transparency = 0  -- Kh√¥ng trong su·ªët
                end
            end
        end
    end
end

-- N√∫t b·∫≠t/t·∫Øt hitbox
AddButton(Tab1o, {

local autoFarmEnabled = false -- Tr·∫°ng th√°i b·∫≠t/t·∫Øt Auto Farm

-- üåü H√†m t√¨m v√† di chuy·ªÉn ƒë·∫øn coin g·∫ßn nh·∫•t
local function collectCoin()
    local player = game.Players.LocalPlayer
    if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then return end

    local closestCoin = nil
    local closestDistance = math.huge

    -- T√¨m coin trong Workspace
    for _, obj in pairs(game.Workspace:GetChildren()) do
        if obj:IsA("Part") and obj.Name:lower():find("coin") then -- Ch·ªâ t√¨m object c√≥ t√™n ch·ª©a "coin"
            local distance = (player.Character.HumanoidRootPart.Position - obj.Position).Magnitude
            if distance < closestDistance then
                closestDistance = distance
                closestCoin = obj
            end
        end
    end

    -- N·∫øu t√¨m th·∫•y coin, di chuy·ªÉn ƒë·∫øn ƒë√≥
    if closestCoin then
        player.Character.HumanoidRootPart.CFrame = closestCoin.CFrame + Vector3.new(0, 3, 0) -- D·ªãch l√™n 3 ƒë·ªÉ tr√°nh l·ªói
    end
end

-- üåü H√†m ch·∫°y Auto Farm
local function startAutoFarm()
    while autoFarmEnabled do
        collectCoin()
        wait(0.5) -- Gi·ªØ kho·∫£ng c√°ch th·ªùi gian ƒë·ªÉ tr√°nh b·ªã ph√°t hi·ªán
    end
end

-- üåü N√∫t b·∫≠t/t·∫Øt Auto Farm

    Callback = function()

        if autoFarmEnabled then
            print("Auto Farm Coin: B·∫¨T ‚úÖ")
            startAutoFarm()
        else
            print("Auto Farm Coin: T·∫ÆT ‚ùå")
        end
    end
})
