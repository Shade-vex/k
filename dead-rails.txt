local attackEnabled = false
local attackRate = 1  -- Mỗi giây tấn công 1 lần
local damageAmount = 50 -- Sát thương mỗi đòn đánh
local attackRange = 1000 -- Bán kính tấn công

-- **🔹 Tìm quái hợp lệ**
local function isValidEnemy(entity)
    if entity:IsA("Model") and entity:FindFirstChild("Humanoid") and entity:FindFirstChild("HumanoidRootPart") then
        -- Không tấn công người chơi
        if game.Players:GetPlayerFromCharacter(entity) then
            return false
        end
        return true
    end
    return false
end

-- **🔹 Kiểm tra khoảng cách**
local function isInRange(entity)
    local player = game.Players.LocalPlayer
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        local distance = (player.Character.HumanoidRootPart.Position - entity.HumanoidRootPart.Position).Magnitude
        return distance <= attackRange
    end
    return false
end

-- **🔹 Gửi lệnh đánh thật đến server**
local function attackEnemies()
    while attackEnabled do
        pcall(function()
            for _, entity in pairs(game.Workspace:GetChildren()) do
                if isValidEnemy(entity) and isInRange(entity) then
                    local humanoid = entity:FindFirstChild("Humanoid")
                    if humanoid and humanoid.Health > 0 then
                        -- **Gửi lệnh tấn công đến server (chính xác)**
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("DamageEnemy", entity, damageAmount)

                        -- **Làm quái mất HP thật**
                        humanoid.Health = humanoid.Health - damageAmount
                    end
                end
            end
        end)
        wait(attackRate)
    end
end

-- **🔹 Bật/Tắt Kill Aura**
local function toggleKillAura()
    attackEnabled = not attackEnabled
    if attackEnabled then
        print("🔥 Kill Aura: BẬT (50 HP/s, 1000m)")
        attackEnemies()
    else
        print("❌ Kill Aura: TẮT")
    end
end

    Name = "⚔️KILL AURA",

local speedEnabled = false
local speedValue = 75  -- Tốc độ di chuyển mong muốn
local defaultSpeed = 16 -- Tốc độ mặc định của game

-- **🔹 Hàm thiết lập tốc độ**
local function setSpeed(value)
    local player = game.Players.LocalPlayer
    if player and player.Character and player.Character:FindFirstChild("Humanoid") then
        player.Character.Humanoid.WalkSpeed = value
    end
end

-- **🔹 Giữ tốc độ không bị reset**
local function maintainSpeed()
    while speedEnabled do
        pcall(function()
            local player = game.Players.LocalPlayer
            if player and player.Character and player.Character:FindFirstChild("Humanoid") then
                local humanoid = player.Character.Humanoid
                if humanoid.WalkSpeed ~= speedValue then
                    humanoid.WalkSpeed = speedValue
                end
            end
        end)
        wait(0.1)  -- Kiểm tra mỗi 0.1 giây để giữ tốc độ
    end
end

-- **🔹 Reset tốc độ nếu chết**
game.Players.LocalPlayer.CharacterAdded:Connect(function()
    if speedEnabled then
        wait(1) -- Đợi nhân vật load xong
        setSpeed(speedValue)
        maintainSpeed()
    end
end)

-- **🔹 Bật/Tắt tăng tốc**
local function toggleSpeed()
    speedEnabled = not speedEnabled
    if speedEnabled then
        print("🚀 Tốc độ: 75 (Bypass Anti-Cheat)")
        setSpeed(speedValue)
        maintainSpeed()
    else
        print("❌ Tốc độ về mặc định")
        setSpeed(defaultSpeed)
    end
end

local espEnabled = false
local espColor = Color3.fromRGB(0, 255, 0) -- Màu xanh lá
local textSize = 15 -- Kích thước chữ cố định
local updateRate = 1 -- Cập nhật mỗi 1 giây
local maxDistance = 1500 -- Bán kính 1500m

-- **🔹 Xác định item có thể nhặt**
local function isValidItem(item)
    if item:IsA("Tool") or item:IsA("Accessory") or item:IsA("BackpackItem") then
        return true
    end
    -- **Chặn các vật thể lớn không phải item**
    local blacklist = {"Tree", "Building", "Wall", "Track", "Bridge", "Platform"}
    for _, name in pairs(blacklist) do
        if item.Name:lower():find(name:lower()) then
            return false
        end
    end
    return false
end

-- **🔹 Hàm tạo ESP cho item**
local function createESP(item)
    local basePart = item:FindFirstChildWhichIsA("BasePart") or item
    if basePart and not basePart:FindFirstChild("EspLabel") then
        local billboard = Instance.new("BillboardGui")
        billboard.Name = "EspLabel"
        billboard.Adornee = basePart
        billboard.Size = UDim2.new(0, 200, 0, 50)
        billboard.StudsOffset = Vector3.new(0, 2, 0)
        billboard.AlwaysOnTop = true

        local textLabel = Instance.new("TextLabel", billboard)
        textLabel.Size = UDim2.new(1, 0, 1, 0)
        textLabel.BackgroundTransparency = 1
        textLabel.TextColor3 = espColor
        textLabel.TextSize = textSize
        textLabel.Font = Enum.Font.SourceSansBold
        textLabel.TextStrokeTransparency = 0.5
        textLabel.TextScaled = false

        billboard.Parent = basePart
    end
end

-- **🔹 Cập nhật ESP cho toàn bộ item mỗi 5 giây**
local function updateESP()
    while espEnabled do
        pcall(function()
            local player = game.Players.LocalPlayer
            if player and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                local playerPos = player.Character.HumanoidRootPart.Position

                for _, item in pairs(game.Workspace:GetChildren()) do
                    if isValidItem(item) then
                        local basePart = item:FindFirstChildWhichIsA("BasePart") or item
                        if basePart then
                            local distance = (playerPos - basePart.Position).Magnitude
                            if distance <= maxDistance then
                                createESP(basePart)
                                local label = basePart:FindFirstChild("EspLabel")
                                if label then
                                    local textLabel = label:FindFirstChildOfClass("TextLabel")
                                    if textLabel then
                                        textLabel.Text = string.format("%s\n[%dm]", item.Name, distance)
                                    end
                                end
                            else
                                if basePart:FindFirstChild("EspLabel") then
                                    basePart.EspLabel:Destroy()
                                end
                            end
                        end
                    end
                end
            end
        end)
        wait(updateRate)
    end
end

-- **🔹 Bật/Tắt ESP**
local function toggleESP()
    espEnabled = not espEnabled
    if espEnabled then
        print("🟢 ESP Item: BẬT (Cập nhật mỗi 5 giây)")
        updateESP()
    else
        print("🔴 ESP Item: TẮT")
        for _, item in pairs(game.Workspace:GetChildren()) do
            if isValidItem(item) then
                local basePart = item:FindFirstChildWhichIsA("BasePart") or item
                if basePart and basePart:FindFirstChild("EspLabel") then
                    basePart.EspLabel:Destroy()
                end
            end
        end
    end
end

-- **🔹 Thêm nút vào menu**
AddButton(Tab3o, {
    Name = "🔰ESP ITEM",
    Callback = function()
        toggleESP()
    end
})








local espEnabled = false -- Trạng thái bật/tắt ESP

local function createESP(player)
    if player == game.Players.LocalPlayer then return end -- Không hiển thị bản thân

    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = character.HumanoidRootPart
    billboard.Size = UDim2.new(0, 200, 0, 50) -- Kích thước mặc định lớn
    billboard.StudsOffset = Vector3.new(0, 3, 0) -- Đặt ở giữa nhân vật
    billboard.AlwaysOnTop = true

    local textLabel = Instance.new("TextLabel", billboard)
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3 = Color3.new(1, 0, 0) -- Màu đỏ
    textLabel.TextScaled = false
    textLabel.TextSize = 20 -- Giữ nguyên kích thước chữ
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.TextStrokeTransparency = 0.2 -- Viền chữ giúp dễ nhìn hơn

    billboard.Parent = character.HumanoidRootPart

    -- Cập nhật vị trí & giữ nguyên kích thước theo khoảng cách
    spawn(function()
        while espEnabled and character and character:FindFirstChild("Humanoid") and character:FindFirstChild("HumanoidRootPart") do
            local humanoid = character:FindFirstChild("Humanoid")
            local localPlayer = game.Players.LocalPlayer
            if localPlayer and localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local distance = (localPlayer.Character.HumanoidRootPart.Position - character.HumanoidRootPart.Position).Magnitude
                
                -- Giữ nguyên kích thước 15 dù ở xa hay gần
                local size = 15
                billboard.Size = UDim2.new(0, size * 10, 0, size * 2)

                -- Hiển thị thông tin máu & khoảng cách
                textLabel.Text = string.format("[%s] HP: %d | %dm", player.Name, math.floor(humanoid.Health), math.floor(distance))
            end
            wait(0.1)
        end
        billboard:Destroy()
    end)
end

-- Hàm bật/tắt ESP
local function toggleESP()
    espEnabled = not espEnabled

    if espEnabled then
        -- Thêm ESP cho tất cả người chơi
        for _, player in pairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer then
                createESP(player)
            end
        end

        -- Lắng nghe người chơi mới tham gia
        game.Players.PlayerAdded:Connect(function(player)
            if espEnabled then createESP(player) end
        end)
    else
        -- Xóa toàn bộ ESP khi tắt
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                for _, gui in pairs(player.Character.HumanoidRootPart:GetChildren()) do
                    if gui:IsA("BillboardGui") then gui:Destroy() end
                end
            end
        end
    end
end

-- Nút bật/tắt ESP
AddButton(Tab3o, {
    Name = "😖ESP PLAYER",
    Callback = function()
        toggleESP()
    end
})
