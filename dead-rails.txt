local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:FindFirstChild("Humanoid")

local godModeEnabled = false -- Biến kiểm soát chế độ bất tử

AddButton(Tab1o, {
    Name = "NEVER DIE",
    Callback = function()
        godModeEnabled = not godModeEnabled -- Đảo trạng thái bật/tắt
        if godModeEnabled then
            print("God Mode ON")

            -- Chặn tất cả sát thương
            game:GetService("RunService").Stepped:Connect(function()
                if humanoid and godModeEnabled then
                    humanoid.Health = humanoid.MaxHealth -- Giữ máu đầy
                    humanoid:ChangeState(11) -- Chặn chết vật lý (fall damage, void damage)
                end
            end)

            -- Chặn chết do scripts khác
            humanoid.BreakJointsOnDeath = false
            character.AncestryChanged:Connect(function()
                if character.Parent == nil and godModeEnabled then
                    player.Character = character -- Ngăn chặn bị xóa nhân vật
                end
            end)
        else
            print("God Mode OFF")
        end
    end
})
