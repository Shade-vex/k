local function FakeDamage()
    if not attackLoopRunning then return end

    local player = game.Players.LocalPlayer
    local character = player.Character
    if not character then return end

    -- Láº¥y vÅ© khÃ­ hiá»‡n táº¡i
    local weapon = character:FindFirstChildOfClass("Tool")
    if not weapon then return end

    -- Kiá»ƒm tra náº¿u vÅ© khÃ­ Ä‘Ãºng loáº¡i Ä‘Ã£ chá»n
    local validWeapon = false
    if SelectedWeapon == "Melee (Cáº­n chiáº¿n)" and weapon.ToolTip == "Melee" then
        validWeapon = true
    elseif SelectedWeapon == "Sword (Kiáº¿m)" and weapon.ToolTip == "Sword" then
        validWeapon = true
    elseif SelectedWeapon == "Blox Fruit (TrÃ¡i Ã¡c quá»·)" and weapon.ToolTip == "Blox Fruit" then
        validWeapon = true
    end

    if not validWeapon then return end

    -- **TÃ¬m quÃ¡i gáº§n nháº¥t trong bÃ¡n kÃ­nh 50m**
    local closestEnemy = nil
    local closestDistance = 50
    for _, enemy in pairs(game.Workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
            local distance = (character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if distance < closestDistance then
                closestDistance = distance
                closestEnemy = enemy
            end
        end
    end

    -- **Táº¥n cÃ´ng quÃ¡i gáº§n nháº¥t**
    if closestEnemy then
        -- **MÃ´ phá»ng nháº¥n chuá»™t**
        game:GetService("VirtualInputManager"):SendMouseButtonEvent(0, 0, 0, true, game, 0)
        wait(0.05) -- Äá»™ trá»… nhá» Ä‘á»ƒ trÃ¡nh spam
        game:GetService("VirtualInputManager"):SendMouseButtonEvent(0, 0, 0, false, game, 0)

        -- **Gá»­i tÃ­n hiá»‡u Ä‘Ã¡nh Ä‘áº¿n server**
        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Combat", weapon.Name, closestEnemy)
    end
end

-- HÃ m tÃ¬m rÆ°Æ¡ng gáº§n nháº¥t
local function getClosestChest()
    local closestChest = nil
    local shortestDistance = math.huge
    if not HumanoidRootPart then return nil end

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") and (obj.Name == "Chest1" or obj.Name == "Chest2" or obj.Name == "Chest3") then
            local distance = (HumanoidRootPart.Position - obj.Position).Magnitude
            if distance < shortestDistance then
                closestChest = obj
                shortestDistance = distance
            end
        end
    end
    return closestChest
end

-- HÃ m di chuyá»ƒn Ä‘áº¿n rÆ°Æ¡ng
local function moveToChest(chest)
    if not chest or not HumanoidRootPart then return end

    -- Báº­t xuyÃªn tÆ°á»ng
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = false
        end
    end

    -- Tween di chuyá»ƒn
    local tween = TweenService:Create(HumanoidRootPart, TweenInfo.new((HumanoidRootPart.Position - chest.Position).Magnitude / speed, Enum.EasingStyle.Linear), {CFrame = chest.CFrame})
    tween:Play()
    tween.Completed:Wait()

    -- Táº¯t xuyÃªn tÆ°á»ng khi nháº·t rÆ°Æ¡ng
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = true
        end
    end
end

-- HÃ m farm rÆ°Æ¡ng
local function farmChests()
    while _G.FarmChest do

        if chest then
            moveToChest(chest)
        end
        wait(0.5) -- Giáº£m táº£i xá»­ lÃ½
    end
end

local function flyToFruit(fruit)
    if not fruit or not fruit:FindFirstChild("Handle") then return end
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end

    toggleNoClip(true) -- Báº­t xuyÃªn tÆ°á»ng

    -- Bay Ä‘áº¿n trÃ¡i vá»›i tá»‘c Ä‘á»™ 300
    local hrp = char.HumanoidRootPart

-- HÃ m tÃ¬m trÃ¡i gáº§n nháº¥t vÃ  nháº·t
local function autoCollect()
    while autoCollectFruits do
        local nearestFruit
        local minDist = math.huge
        local playerPos = player.Character.HumanoidRootPart.Position

        -- TÃ¬m trÃ¡i gáº§n nháº¥t
        for _, fruit in pairs(game.Workspace:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                local fruitPos = fruit.Handle.Position
                local dist = (playerPos - fruitPos).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestFruit = fruit
                end
            end
        end

        -- Náº¿u tÃ¬m tháº¥y trÃ¡i, bay Ä‘áº¿n nháº·t
        if nearestFruit then
            flyToFruit(nearestFruit)
        end

        wait(1) -- Kiá»ƒm tra má»—i giÃ¢y
    end
end

-- HÃ m báº­t/táº¯t Auto Nháº·t TrÃ¡i

local function setHitbox(player)
    -- Kiá»ƒm tra khÃ´ng Ã¡p dá»¥ng hitbox cho chÃ­nh mÃ¬nh
    if player == game.Players.LocalPlayer then return end  

    local character = player.Character
