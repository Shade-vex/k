-- **H√†m t√¨m boss g·∫ßn nh·∫•t**
local function getNearestBoss()
    local closestBoss = nil
    local closestDistance = math.huge
    for _, enemy in pairs(Workspace:GetChildren()) do
        if enemy:IsA("Model") and enemy:FindFirstChild("Humanoid") and enemy:FindFirstChild("HumanoidRootPart") then
            local distance = (LocalPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if distance < closestDistance then
                closestBoss = enemy
                closestDistance = distance
            end
        end
    end
    return closestBoss
end

-- **H√†m farm boss**
local function farmBoss()
    bossFarmEnabled = not bossFarmEnabled
    while bossFarmEnabled do
        local boss = getNearestBoss()
        if boss and boss:FindFirstChild("HumanoidRootPart") then
            -- Bay tr√™n ƒë·∫ßu boss
            LocalPlayer.Character.HumanoidRootPart.CFrame = boss.HumanoidRootPart.CFrame * CFrame.new(0, 10, 0)
            -- TƒÉng hitbox boss x10 l·∫ßn
            boss.HumanoidRootPart.Size = Vector3.new(50, 50, 50)
            -- T·ª± ƒë·ªông ƒë√°nh
            if LocalPlayer.Character:FindFirstChildOfClass("Tool") then
                LocalPlayer.Character:FindFirstChildOfClass("Tool"):Activate()
            end
        end
        wait(0.1) -- Gi·ªØ v·ªã tr√≠ tr√™n ƒë·∫ßu boss
    end
end

local function collectFruit(fruit)
    if not fruit or not fruit:FindFirstChild("Handle") then return end
    local player = game.Players.LocalPlayer
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end

    -- Bay ƒë·∫øn tr√°i (cao h∆°n m·ªôt ch√∫t ƒë·ªÉ tr√°nh l·ªói va ch·∫°m)
    char.HumanoidRootPart.CFrame = fruit.Handle.CFrame + Vector3.new(0, 3, 0)
end

-- H√†m t√¨m tr√°i g·∫ßn nh·∫•t v√† nh·∫∑t
local function autoCollect()
    while autoCollectFruits do
        local nearestFruit
        local minDist = math.huge
        local playerPos = game.Players.LocalPlayer.Character.HumanoidRootPart.Position

        -- T√¨m tr√°i g·∫ßn nh·∫•t
        for _, fruit in pairs(game.Workspace:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                local fruitPos = fruit.Handle.Position
                local dist = (playerPos - fruitPos).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestFruit = fruit
                end
            end
        end

        -- N·∫øu t√¨m th·∫•y tr√°i, di chuy·ªÉn ƒë·∫øn nh·∫∑t
        if nearestFruit then
            collectFruit(nearestFruit)
        end

        wait(1) -- Ki·ªÉm tra m·ªói gi√¢y
    end
end

-- H√†m b·∫≠t/t·∫Øt Auto Nh·∫∑t Tr√°i
local function toggleAutoCollect()
    autoCollectFruits = not autoCollectFruits

    if autoCollectFruits then
        autoCollect()
    end
end

            for _, part in pairs(LocalPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                end
            end
