local Player = game.Players.LocalPlayer

_G.FarmChest = false
local speed = 300 -- T·ªëc ƒë·ªô bay c·ªë ƒë·ªãnh

-- H√†m t√¨m r∆∞∆°ng g·∫ßn nh·∫•t
local function getClosestChest()
    local closestChest = nil
    local shortestDistance = math.huge
    if not HumanoidRootPart then return nil end

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") and (obj.Name == "Chest1" or obj.Name == "Chest2" or obj.Name == "Chest3") then
            local distance = (HumanoidRootPart.Position - obj.Position).Magnitude
            if distance < shortestDistance then
                closestChest = obj
                shortestDistance = distance
            end
        end
    end
    return closestChest
end

-- H√†m di chuy·ªÉn ƒë·∫øn r∆∞∆°ng
local function moveToChest(chest)
    if not chest or not HumanoidRootPart then return end

    -- B·∫≠t xuy√™n t∆∞·ªùng
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = false
        end
    end

    -- Tween di chuy·ªÉn
    local tween = TweenService:Create(HumanoidRootPart, TweenInfo.new((HumanoidRootPart.Position - chest.Position).Magnitude / speed, Enum.EasingStyle.Linear), {CFrame = chest.CFrame})
    tween:Play()
    tween.Completed:Wait()

    -- T·∫Øt xuy√™n t∆∞·ªùng khi nh·∫∑t r∆∞∆°ng
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = true
        end
    end
end

-- H√†m farm r∆∞∆°ng
local function farmChests()
    while _G.FarmChest do
        local chest = getClosestChest()
        if chest then
            moveToChest(chest)
        end
        wait(0.5) -- Gi·∫£m t·∫£i x·ª≠ l√Ω
    end
end

local autoCollectFruits = false -- Bi·∫øn b·∫≠t/t·∫Øt Auto Nh·∫∑t Tr√°i
local player = game.Players.LocalPlayer
local tweenService = game:GetService("TweenService")

-- H√†m b·∫≠t/t·∫Øt xuy√™n t∆∞·ªùng
local function toggleNoClip(state)
    for _, part in pairs(player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = not state
        end
    end
end

-- H√†m bay ƒë·∫øn tr√°i c√¢y
local function flyToFruit(fruit)
    if not fruit or not fruit:FindFirstChild("Handle") then return end
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end

    toggleNoClip(true) -- B·∫≠t xuy√™n t∆∞·ªùng

    -- Bay ƒë·∫øn tr√°i v·ªõi t·ªëc ƒë·ªô 300
    local hrp = char.HumanoidRootPart
    local goalCFrame = fruit.Handle.CFrame + Vector3.new(0, 3, 0)
    local tween = tweenService:Create(hrp, TweenInfo.new((hrp.Position - fruit.Handle.Position).Magnitude / 300, Enum.EasingStyle.Linear), {CFrame = goalCFrame})

    tween:Play()
    tween.Completed:Wait()

    toggleNoClip(false) -- T·∫Øt xuy√™n t∆∞·ªùng khi ƒë·∫øn n∆°i
end

-- H√†m t√¨m tr√°i g·∫ßn nh·∫•t v√† nh·∫∑t
local function autoCollect()
    while autoCollectFruits do
        local nearestFruit
        local minDist = math.huge
        local playerPos = player.Character.HumanoidRootPart.Position

        -- T√¨m tr√°i g·∫ßn nh·∫•t
        for _, fruit in pairs(game.Workspace:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                local fruitPos = fruit.Handle.Position
                local dist = (playerPos - fruitPos).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestFruit = fruit
                end
            end
        end

        -- N·∫øu t√¨m th·∫•y tr√°i, bay ƒë·∫øn nh·∫∑t
        if nearestFruit then
            flyToFruit(nearestFruit)
        end

        wait(1) -- Ki·ªÉm tra m·ªói gi√¢y
    end
end

-- H√†m b·∫≠t/t·∫Øt Auto Nh·∫∑t Tr√°i
local function toggleAutoCollect()
    autoCollectFruits = not autoCollectFruits

    if autoCollectFruits then
        autoCollect()
    end
end

            local tween = tweenService:Create(hrp, tweenInfo, {CFrame = targetHrp.CFrame})

            tween:Play()

            for _, part in pairs(LocalPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                end
            end
        end
