local Player = game.Players.LocalPlayer
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local HumanoidRootPart = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")

_G.FarmChest = false
local speed = 300 -- Tốc độ bay cố định

-- Hàm tìm rương gần nhất
local function getClosestChest()
    local closestChest = nil
    local shortestDistance = math.huge
    if not HumanoidRootPart then return nil end

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") and (obj.Name == "Chest1" or obj.Name == "Chest2" or obj.Name == "Chest3") then
            local distance = (HumanoidRootPart.Position - obj.Position).Magnitude
            if distance < shortestDistance then
                closestChest = obj
                shortestDistance = distance
            end
        end
    end
    return closestChest
end

-- Hàm di chuyển đến rương
local function moveToChest(chest)
    if not chest or not HumanoidRootPart then return end

    -- Bật xuyên tường
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = false
        end
    end

    -- Tween di chuyển
    local tween = TweenService:Create(HumanoidRootPart, TweenInfo.new((HumanoidRootPart.Position - chest.Position).Magnitude / speed, Enum.EasingStyle.Linear), {CFrame = chest.CFrame})
    tween:Play()
    tween.Completed:Wait()

    -- Tắt xuyên tường khi nhặt rương
    for _, part in ipairs(Player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = true
        end
    end
end

-- Hàm farm rương
local function farmChests()
    while _G.FarmChest do
        local chest = getClosestChest()
        if chest then
            moveToChest(chest)
        end
        wait(0.5) -- Giảm tải xử lý
    end
end

-- Nút bật/tắt farm rương
AddButton(Tab2o, {
    Name = "Farm Chest",
    Callback = function()
        _G.FarmChest = not _G.FarmChest
        if _G.FarmChest then
            spawn(farmChests)
        end
    end
})

local autoCollectFruits = false -- Biến bật/tắt Auto Nhặt Trái
local player = game.Players.LocalPlayer
local tweenService = game:GetService("TweenService")

-- Hàm bật/tắt xuyên tường
local function toggleNoClip(state)
    for _, part in pairs(player.Character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = not state
        end
    end
end

-- Hàm bay đến trái cây
local function flyToFruit(fruit)
    if not fruit or not fruit:FindFirstChild("Handle") then return end
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end

    toggleNoClip(true) -- Bật xuyên tường

    -- Bay đến trái với tốc độ 300
    local hrp = char.HumanoidRootPart
    local goalCFrame = fruit.Handle.CFrame + Vector3.new(0, 3, 0)
    local tween = tweenService:Create(hrp, TweenInfo.new((hrp.Position - fruit.Handle.Position).Magnitude / 300, Enum.EasingStyle.Linear), {CFrame = goalCFrame})

    tween:Play()
    tween.Completed:Wait()

    toggleNoClip(false) -- Tắt xuyên tường khi đến nơi
end

-- Hàm tìm trái gần nhất và nhặt
local function autoCollect()
    while autoCollectFruits do
        local nearestFruit
        local minDist = math.huge
        local playerPos = player.Character.HumanoidRootPart.Position

        -- Tìm trái gần nhất
        for _, fruit in pairs(game.Workspace:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                local fruitPos = fruit.Handle.Position
                local dist = (playerPos - fruitPos).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestFruit = fruit
                end
            end
        end

        -- Nếu tìm thấy trái, bay đến nhặt
        if nearestFruit then
            flyToFruit(nearestFruit)
        end

        wait(1) -- Kiểm tra mỗi giây
    end
end

-- Hàm bật/tắt Auto Nhặt Trái
local function toggleAutoCollect()
    autoCollectFruits = not autoCollectFruits

    if autoCollectFruits then
        autoCollect()
    end
end

-- Nút bật/tắt Auto Nhặt Trái
AddButton(Tab6o, {
    Name = "Auto Nhặt Trái",
    Callback = function()
        toggleAutoCollect()
    end
})

local function teleportToPlayerV2()
    if selectedPlayerV2 and selectedPlayerV2.Character and selectedPlayerV2.Character:FindFirstChild("HumanoidRootPart") then
        local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        local targetHrp = selectedPlayerV2.Character:FindFirstChild("HumanoidRootPart")

        if hrp and targetHrp then
            _G.FlyingV2 = true -- Bật chế độ bay

            -- Xuyên tường khi di chuyển
            for _, part in pairs(LocalPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end

            -- Tween bay đến mục tiêu
            local tweenService = game:GetService("TweenService")
            local tweenInfo = TweenInfo.new((hrp.Position - targetHrp.Position).Magnitude / 300, Enum.EasingStyle.Linear)
            local tween = tweenService:Create(hrp, tweenInfo, {CFrame = targetHrp.CFrame})

            tween:Play()
            tween.Completed:Wait() -- Chờ đến nơi thì dừng bay

            _G.FlyingV2 = false -- Tắt chế độ bay

            -- Khôi phục va chạm sau khi dịch chuyển xong
            for _, part in pairs(LocalPlayer.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                end
            end
        end
    else
        warn("Không thể dịch chuyển! Người chơi không hợp lệ hoặc đã rời game.")
    end
end

-- Vòng tròn AimBot
function updateCircle(Fov, colorCircle)
    if circle.__OBJECT_EXISTS then
        circle.Transparency = 1
        circle.Visible = true
        circle.Thickness = 2
        circle.Color = colorCircle
        circle.NumSides = 100
        circle.Radius = (Fov * 6) / 2
        circle.Filled = false
        circle.Position = Vector2.new(mouse.X, mouse.Y + (guiservice:GetGuiInset().Y))
    end
end

-- Theo dõi vị trí kẻ địch trong phạm vi FOV
spawn(function()
    game:GetService("RunService").RenderStepped:Connect(function()
        if AimBotEnabled then
            pcall(function()
                updateCircle(getgenv().setting['Fov'], getgenv().setting['Color'])
            end)
        else
            circle.Visible = false
        end
    end)
end)

-- Xác định mục tiêu
spawn(function()
    while wait() do
        if AimBotEnabled then
            for i, v in pairs(players:GetPlayers()) do
                if v ~= localPlayer and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                    local pos = currentCamera:WorldToViewportPoint(v.Character.HumanoidRootPart.Position)

local hitboxEnabled = false  -- Biến để bật/tắt hitbox
local hitboxSize = Vector3.new(70, 70, 70)  -- Kích thước hitbox x70
local transparency = 0.5  -- Độ trong suốt của hitbox (0 là không trong suốt, 1 là vô hình)
local hitboxLoopRunning = false  -- Biến kiểm soát vòng lặp kiểm tra hitbox

-- Hàm chỉnh hitbox của người chơi khác
local function setHitbox(player)
    -- Kiểm tra không áp dụng hitbox cho chính mình
    if player == game.Players.LocalPlayer then return end  

    local character = player.Character
    if character and character:FindFirstChild("HumanoidRootPart") then
        local rootPart = character.HumanoidRootPart
        rootPart.Size = hitboxSize
        rootPart.Transparency = transparency
        rootPart.CanCollide = false
    end
end

-- Hàm kiểm tra và cập nhật hitbox cho tất cả người chơi trong game
local function checkAndSetHitboxes()
    if hitboxEnabled then
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character then
                -- Kiểm tra và set hitbox cho player
                setHitbox(player)
            end
        end
    end
end

-- Hiển thị chữ "ValorVex Hitbox V1" bảy màu trong 3 giây
local function showRainbowText()
    local gui = Instance.new("ScreenGui", game.Players.LocalPlayer.PlayerGui)
    local label = Instance.new("TextLabel", gui)

    label.Size = UDim2.new(0.4, 0, 0.1, 0)  -- Kích thước chữ
    label.Position = UDim2.new(0.3, 0, 0.4, 0)  -- Giữa màn hình
    label.Text = "ValorVex Hitbox V1"
    label.Font = Enum.Font.SourceSansBold
    label.TextScaled = true
    label.BackgroundTransparency = 1
    label.TextStrokeTransparency = 0
    label.TextColor3 = Color3.new(1, 0, 0)  -- Bắt đầu với màu đỏ

    -- Hiệu ứng đổi màu bảy sắc cầu vồng
    local colors = {
        Color3.new(1, 0, 0),  -- Đỏ
        Color3.new(1, 0.5, 0),  -- Cam
        Color3.new(1, 1, 0),  -- Vàng
        Color3.new(0, 1, 0),  -- Xanh lá
        Color3.new(0, 1, 1),  -- Xanh dương
        Color3.new(0, 0, 1),  -- Xanh đậm
        Color3.new(0.5, 0, 1)  -- Tím
    }

    task.spawn(function()
        for i = 1, 30 do  -- Chạy trong 3 giây (mỗi lần đổi màu 0.1 giây)
            label.TextColor3 = colors[(i % #colors) + 1]
            wait(0.1)
        end
        gui:Destroy()  -- Xóa chữ sau 3 giây
    end)
end

-- Hàm bật/tắt hitbox
local function toggleHitbox()
    hitboxEnabled = not hitboxEnabled
    if hitboxEnabled then
        -- Áp dụng hitbox cho tất cả player hiện tại (trừ LocalPlayer)
        checkAndSetHitboxes()

        -- Kết nối sự kiện khi có player mới vào game và sau 3s sẽ thay đổi hitbox
        game.Players.PlayerAdded:Connect(function(player)
            player.CharacterAdded:Connect(function()
                wait(3)  -- Chờ 3 giây để nhân vật load xong
                setHitbox(player)
            end)
        end)

        -- Kết nối sự kiện khi player hồi sinh
        game.Players.PlayerAdded:Connect(function(player)
            player.CharacterAdded:Connect(function()
                wait(3)  -- Chờ 3 giây để nhân vật load xong
                setHitbox(player)
            end)
        end)

        showRainbowText()  -- Hiển thị chữ khi bật hitbox

        -- Bắt đầu vòng lặp kiểm tra hitbox mỗi giây
        hitboxLoopRunning = true
        while hitboxLoopRunning do
            wait(1)  -- Kiểm tra mỗi giây
            checkAndSetHitboxes()  -- Kiểm tra và set hitbox cho tất cả player
        end
    else
        -- Dừng vòng lặp và reset hitbox cho tất cả player
        hitboxLoopRunning = false
        for _, player in pairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer then
                local character = player.Character
                if character and character:FindFirstChild("HumanoidRootPart") then
                    local rootPart = character.HumanoidRootPart
                    rootPart.Size = Vector3.new(2, 2, 1)  -- Kích thước mặc định
                    rootPart.Transparency = 0  -- Không trong suốt
                end
            end
        end
    end
end

    end

local espEnabled = false -- Trạng thái bật/tắt ESP

local function createESP(player)
    if player == game.Players.LocalPlayer then return end -- Không hiển thị bản thân

    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = character.HumanoidRootPart
    billboard.Size = UDim2.new(0, 200, 0, 50) -- Kích thước mặc định lớn
    billboard.StudsOffset = Vector3.new(0, 3, 0) -- Đặt ở giữa nhân vật
    billboard.AlwaysOnTop = true

    local textLabel = Instance.new("TextLabel", billboard)
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3 = Color3.new(1, 0, 0) -- Màu đỏ
    textLabel.TextScaled = false
    textLabel.TextSize = 20 -- Giữ nguyên kích thước chữ
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.TextStrokeTransparency = 0.2 -- Viền chữ giúp dễ nhìn hơn

    billboard.Parent = character.HumanoidRootPart

    -- Cập nhật vị trí & giữ nguyên kích thước theo khoảng cách
    spawn(function()
        while espEnabled and character and character:FindFirstChild("Humanoid") and character:FindFirstChild("HumanoidRootPart") do
            local humanoid = character:FindFirstChild("Humanoid")
            local localPlayer = game.Players.LocalPlayer
            if localPlayer and localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local distance = (localPlayer.Character.HumanoidRootPart.Position - character.HumanoidRootPart.Position).Magnitude
                
                -- Giữ nguyên kích thước 20 dù ở xa hay gần
                local size = 20
                billboard.Size = UDim2.new(0, size * 10, 0, size * 2)

                -- Hiển thị thông tin máu & khoảng cách
                textLabel.Text = string.format("[%s] HP: %d | %dm", player.Name, math.floor(humanoid.Health), math.floor(distance))
            end
            wait(0.1)
        end
        billboard:Destroy()
    end)
end

-- Hàm bật/tắt ESP
local function toggleESP()
    espEnabled = not espEnabled

    if espEnabled then
        -- Thêm ESP cho tất cả người chơi
        for _, player in pairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer then
                createESP(player)
            end
        end

        -- Lắng nghe người chơi mới tham gia
        game.Players.PlayerAdded:Connect(function(player)
            if espEnabled then createESP(player) end
        end)
    else
        -- Xóa toàn bộ ESP khi tắt
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                for _, gui in pairs(player.Character.HumanoidRootPart:GetChildren()) do
                    if gui:IsA("BillboardGui") then gui:Destroy() end
                end
            end
        end
    end
end

local function getGameLanguage()
    local locale = game.LocalizationService.RobloxLocaleId
    if locale:find("en") then
        return "en" -- Tiếng Anh
    else
        return "vi" -- Tiếng Việt (mặc định)
    end
end

local function updateFruitESP()
    if not espFruitEnabled then return end

    local localPlayer = game.Players.LocalPlayer  
    if not localPlayer or not localPlayer.Character or not localPlayer.Character:FindFirstChild("HumanoidRootPart") then return end  

    local closestFruit = nil  
    local closestDistance = math.huge  

    for _, fruit in pairs(game.Workspace:GetChildren()) do  
        if fruit:IsA("Tool") and fruit:FindFirstChild("Handle") then  
            local distance = (localPlayer.Character.HumanoidRootPart.Position - fruit.Handle.Position).Magnitude  
            if distance < closestDistance then  
                closestFruit = fruit  
                closestDistance = distance  
            end  
        end  
    end  

    -- Xác định ngôn ngữ hiển thị
    local lang = getGameLanguage()

    if closestFruit then  
        if lang == "vi" and fruitLabelVI then
            fruitLabelVI.Text = string.format("Đã phát hiện trái cây: Cách đây %dm", math.floor(closestDistance))

            fruitLabelEN.Text = string.format("FRUIT DETECTED: %dm away.", math.floor(closestDistance))  
            fruitLabelVI.Text = "" -- Ẩn thông báo tiếng Việt
        end
    else  
        if fruitLabelVI then fruitLabelVI.Text = "" end

local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local PlaceId = game.PlaceId

local function switchServer()
    local success, servers = pcall(function()
        return HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"))
    end)

    if success and servers and servers.data then
        for _, server in pairs(servers.data) do
            if server.id ~= game.JobId and server.playing < server.maxPlayers then
                TeleportService:TeleportToPlaceInstance(PlaceId, server.id, game.Players.LocalPlayer)
                return
            end
        end
    end
end

local Player = game.Players.LocalPlayer
local Character = Player.Character or Player.CharacterAdded:Wait()
local workspace = game:GetService("Workspace")

-- Bán kính để kiểm tra các thực thể
local radius = 5000000
local auraActive = false
local damage = 1000000000 -- Số lượng sát thương sẽ được gây ra mỗi giây

-- Danh sách người chơi không bị ảnh hưởng bởi Kill Aura
local excludedPlayers = {
    "n1LK_v033STPE", -- Người chơi này sẽ không bị tác động
    "loc2k2LivesStream",    -- Thêm tên người chơi thứ hai
    "Yrixyakita",     -- Thêm tên người chơi thứ ba
    "rbown2801"      -- Thêm tên người chơi thứ tư
}

-- Hàm để giảm HP của các thực thể có Humanoid trong bán kính
local function applyKillAura()
    while auraActive do
        -- Tìm các đối tượng xung quanh Player trong bán kính 5000000m
        local charactersInRange = {}
        for _, object in pairs(workspace:GetChildren()) do
            if object:IsA("Model") and object:FindFirstChild("Humanoid") then
                local humanoid = object:FindFirstChild("Humanoid")
                local distance = (object.PrimaryPart.Position - Character.PrimaryPart.Position).Magnitude
                if distance <= radius and humanoid.Health > 0 and object ~= Character then
                    table.insert(charactersInRange, object)
                end
            end
        end
        
        -- Duyệt qua các đối tượng và gây sát thương thực sự
        for _, entity in pairs(charactersInRange) do
            local humanoid = entity:FindFirstChild("Humanoid")
            if humanoid and humanoid.Health > 0 then
                -- Kiểm tra xem tên người chơi có trong danh sách không
                local isExcluded = false
                for _, excludedName in ipairs(excludedPlayers) do
                    if entity.Name == excludedName then
                        isExcluded = true
                        break
                    end
                end
                
                -- Nếu người chơi không có trong danh sách, gây sát thương
                if not isExcluded then
                    humanoid:TakeDamage(damage) -- Gây sát thương thực sự

                    -- Nếu HP đã xuống dưới 0, "giết" thực thể
                    if humanoid.Health <= 0 then
                        -- Gọi sự kiện Died
                        humanoid.Health = 0
                        entity:BreakJoints() -- Hủy các khớp để nhân vật chết ngay lập tức
                    end
                end
            end
        end
        
        wait(1) -- Đợi 1 giây trước khi kiểm tra lại
    end
end

-- Hàm để tạo nút bật/tắt Kill Aura
local function toggleKillAura()
    auraActive = not auraActive
    if auraActive then
        -- Khi bật Kill Aura
        spawn(function()
            applyKillAura()
        end)
    end
end

-- Thêm nút bật/tắt Kill Aura vào Tab11o
AddButton(Tab11o, {
    Name = "Haki bá vương",
    Callback = function()
        toggleKillAura()
    end
})
